public boolean add(int index, E e) {
    if (index < 0 || index > size) {
        return false;
    }
    ListNode<E> newNode = new ListNode<>(e);
    if (index == 0) {
        newNode.next = head;
        head = newNode;
    } else {
        ListNode<E> prev = head;
        for (int i = 0; i < index - 1; i++) {
            prev = prev.next;
        }
        newNode.next = prev.next;
        prev.next = newNode;
    }
    size++;
    return true;
}
